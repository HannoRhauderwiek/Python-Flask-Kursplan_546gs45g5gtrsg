Besucherzähler

Aufgabe: Session-basierter Zähler

app.py

from flask import Flask, render_template, session, redirect

app = Flask(__name__)
app.secret_key = 'dein-geheimnis-hier'

@app.route('/zaehler')
def zaehler():
    """Zähler erhöhen und anzeigen"""
    # Session-Daten mit .get() abrufen (Standardwert 0 falls nicht vorhanden)
    session['besuche'] = session.get('besuche', 0) + 1
    return render_template('zaehler.html', besuche=session['besuche'])

@app.route('/zaehler/reset')
def zaehler_reset():
    """Zähler zurücksetzen"""
    session.pop('besuche', None)  # Entfernt 'besuche' aus der Session
    return redirect('/zaehler')

if __name__ == '__main__':
    app.run(debug=True)

Wie es funktioniert: - session.get('besuche', 0) gibt die aktuelle Anzahl zurück (oder 0) - Bei jedem Besuch wird die Zahl um 1 erhöht - session.pop() entfernt einen Wert aus der Session - Die Session bleibt auch nach Neuladen der Seite erhalten!

Zahlenratespiel

Aufgabe: Komplettes Ratespiel mit Session

app.py

from flask import Flask, render_template, request, session, redirect
import random

app = Flask(__name__)
app.secret_key = 'dein-geheimnis-hier'

@app.route('/')
def index():
    """Startseite und Spielanzeige"""
    # Neues Spiel initialisieren wenn nötig
    if 'zahl' not in session:
        session['zahl'] = random.randint(1, 100)
        session['versuche'] = 0
        session['gewonnen'] = False
        session['highscore'] = session.get('highscore', float('inf'))

    return render_template('spiel.html', session_data=session)

@app.route('/rate', methods=['POST'])
def rate():
    """Tipp verarbeiten"""
    if session.get('gewonnen'):
        return redirect('/')

    try:
        tipp = int(request.form.get('tipp', 0))
    except ValueError:
        return redirect('/')

    if tipp < 1 or tipp > 100:
        return redirect('/')

    session['versuche'] += 1
    zahl = session['zahl']
    unterschied = abs(tipp - zahl)

    # Bestimme Rückmeldung
    if tipp == zahl:
        session['gewonnen'] = True
        # Neuer Highscore?
        if session['versuche'] < session.get('highscore', float('inf')):
            session['highscore'] = session['versuche']
    elif tipp < zahl:
        session['feedback'] = f"❌ {tipp} ist zu niedrig!"
    else:
        session['feedback'] = f"❌ {tipp} ist zu hoch!"

    # "Nah dran" Hinweis
    if unterschied < 10 and unterschied > 0:
        session['feedback'] += " 🔥 Du bist nah dran!"

    return redirect('/')

@app.route('/neues-spiel', methods=['POST'])
def neues_spiel():
    """Neues Spiel starten"""
    session['zahl'] = random.randint(1, 100)
    session['versuche'] = 0
    session['gewonnen'] = False
    session['feedback'] = None
    return redirect('/')

if __name__ == '__main__':
    app.run(debug=True)

Besonderheiten: - Random Zahl: Wird nur beim Spielstart oder Neustarten generiert - Highscore Tracking: Wird in der Session gespeichert - “Nah dran” Hinweis: Wenn Unterschied < 10 - Fehlerbehandlung: Nur Zahlen 1-100 werden akzeptiert - POST für Aktionen: Tipps werden mit POST gesendet (sicherer) - Auto-Focus: Input-Feld ist sofort fokussiert

To-Do-App

Aufgabe: Komplette To-Do-Anwendung mit Datenbankfilter

schema.sql

CREATE TABLE todos (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    aufgabe TEXT NOT NULL,
    erledigt INTEGER DEFAULT 0,
    erstellt TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

init_db.py

import sqlite3

def init_db(db_name='todos.db'):
    conn = sqlite3.connect(db_name)
    cursor = conn.cursor()

    # Tabelle erstellen
    with open('schema.sql', 'r') as f:
        cursor.executescript(f.read())

    # Testdaten
    testaufgaben = [
        ('Python lernen', 1),
        ('Flask-Projekt abschließen', 1),
        ('Datenbank designen', 0),
        ('Tests schreiben', 0),
        ('Dokumentation aktualisieren', 0),
    ]

    cursor.executemany(
        'INSERT INTO todos (aufgabe, erledigt) VALUES (?, ?)',
        testaufgaben
    )

    conn.commit()
    conn.close()
    print(f"✓ To-Do-Datenbank mit {len(testaufgaben)} Aufgaben erstellt!")

if __name__ == '__main__':
    init_db()

app.py

from flask import Flask, render_template, request, redirect, flash
import sqlite3

app = Flask(__name__)
app.secret_key = 'dein-geheimnis-hier'

def get_db_connection():
    conn = sqlite3.connect('todos.db')
    conn.row_factory = sqlite3.Row
    return conn

@app.route('/')
def todo_liste():
    """Zeige To-Do-Liste mit Filter"""
    filter_type = request.args.get('filter', 'alle')

    conn = get_db_connection()
    cursor = conn.cursor()

    # SQL basierend auf Filter ändern
    if filter_type == 'offen':
        cursor.execute('SELECT * FROM todos WHERE erledigt = 0 ORDER BY erstellt DESC')
    elif filter_type == 'erledigt':
        cursor.execute('SELECT * FROM todos WHERE erledigt = 1 ORDER BY erstellt DESC')
    else:  # 'alle'
        cursor.execute('SELECT * FROM todos ORDER BY erstellt DESC')

    todos = cursor.fetchall()

    # Statistiken berechnen
    cursor.execute('SELECT COUNT(*) FROM todos')
    gesamt = cursor.fetchone()[0]

    cursor.execute('SELECT COUNT(*) FROM todos WHERE erledigt = 1')
    fertig = cursor.fetchone()[0]

    cursor.execute('SELECT COUNT(*) FROM todos WHERE erledigt = 0')
    offen = cursor.fetchone()[0]

    conn.close()

    return render_template(
        'todo_liste.html',
        todos=todos,
        filter=filter_type,
        gesamt=gesamt,
        fertig=fertig,
        offen=offen
    )

@app.route('/neu', methods=['GET', 'POST'])
def todo_neu():
    """Neue To-Do hinzufügen"""
    if request.method == 'POST':
        aufgabe = request.form.get('aufgabe', '').strip()

        # Validierung
        if not aufgabe:
            flash('⚠️ Aufgabe darf nicht leer sein!', 'warning')
            return redirect('/neu')

        if len(aufgabe) < 3:
            flash('⚠️ Aufgabe muss mindestens 3 Zeichen lang sein!', 'warning')
            return redirect('/neu')

        # In Datenbank speichern
        conn = get_db_connection()
        cursor = conn.cursor()
        cursor.execute('INSERT INTO todos (aufgabe) VALUES (?)', (aufgabe,))
        conn.commit()
        conn.close()

        flash('✅ Aufgabe hinzugefügt!', 'success')
        return redirect('/')

    return render_template('todo_form.html')

@app.route('/toggle/<int:id>', methods=['POST'])
def todo_toggle(id):
    """To-Do als erledigt/unerledigt markieren"""
    conn = get_db_connection()
    cursor = conn.cursor()

    # Aktuellen Status abrufen
    cursor.execute('SELECT erledigt FROM todos WHERE id = ?', (id,))
    result = cursor.fetchone()

    if result:
        neuer_status = 1 - result['erledigt']  # Toggle: 0→1 oder 1→0
        cursor.execute('UPDATE todos SET erledigt = ? WHERE id = ?', (neuer_status, id))
        conn.commit()

    conn.close()
    return redirect('/')

@app.route('/loeschen/<int:id>', methods=['POST'])
def todo_loeschen(id):
    """To-Do löschen"""
    conn = get_db_connection()
    cursor = conn.cursor()

    cursor.execute('DELETE FROM todos WHERE id = ?', (id,))
    conn.commit()
    conn.close()

    flash('🗑️ Aufgabe gelöscht!', 'info')
    return redirect('/')

if __name__ == '__main__':
    app.run(debug=True)

Features der To-Do-App: - Filter: Alle / Offen / Erledigt - filtern direkt in der Datenbank - Checkboxes: ☐/☑ zum schnellen Toggle ohne eigene Seite - Strikethrough: Erledigte Aufgaben visuell unterscheiden - Statistiken: Live-Zähler auf der Hauptseite - Datumsspeicherung: Automatic CURRENT_TIMESTAMP - Löschen mit Bestätigung: JavaScript confirm() - Validierung: Aufgabe muss mindestens 3 Zeichen sein - Bootstrap-Styling: Moderne, responsive UI mit List Groups

Session vs. Datenbank: Wann was?

Session verwenden für:

  • Besucherzähler: Pro Benutzer, vergessen nach Session-Ende ✓
  • Zahlenratespiel: Spielzustand, Highscore in Session speichern ✓
  • Einkaufswagen: Temporäre Daten, nicht persistent
  • Login-Status: Wer ist angemeldet?

Datenbank verwenden für:

  • To-Do-Liste: Langfristige Speicherung, alle Benutzer ✓
  • Gästebuch: Alle Einträge sichtbar, permanent
  • Benutzerprofile: Persistente Benutzerdaten
  • Anything “wichtig”: Sicherheit, Permanenz!

Merksatz: Session = kurzzeitig & privat | Datenbank = dauerhaft & öffentlich
