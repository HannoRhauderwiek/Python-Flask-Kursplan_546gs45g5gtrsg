Bücherverwaltung mit SQL (pure Python)

Aufgabe: buecher_sql.py

import sqlite3

# Verbindung zur Datenbank herstellen
conn = sqlite3.connect('buecher.db')
cursor = conn.cursor()

# Tabelle erstellen
cursor.execute('''
    CREATE TABLE IF NOT EXISTS buecher (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        titel TEXT NOT NULL,
        autor TEXT NOT NULL,
        jahr INTEGER,
        bewertung REAL
    )
''')

# 5 Bücher einfügen
buecher = [
    ('Der Hobbit', 'J.R.R. Tolkien', 1937, 4.8),
    ('Harry Potter und der Stein der Weisen', 'J.K. Rowling', 1997, 4.9),
    ('1984', 'George Orwell', 1949, 4.6),
    ('Die Welle', 'Todd Strasser', 1981, 4.3),
    ('Tschick', 'Wolfgang Herrndorf', 2010, 4.7)
]

cursor.executemany(
    'INSERT INTO buecher (titel, autor, jahr, bewertung) VALUES (?, ?, ?, ?)',
    buecher
)
conn.commit()

print("=== ALLE BÜCHER ===")
cursor.execute('SELECT * FROM buecher')
for buch in cursor.fetchall():
    print(f"{buch[0]}. {buch[1]} - {buch[2]} ({buch[3]} ⭐)")

print("\n=== BÜCHER NACH 2000 ===")
cursor.execute('SELECT titel, author, jahr FROM buecher WHERE jahr > 2000 ORDER BY jahr DESC')
for titel, autor, jahr in cursor.fetchall():
    print(f"{titel} ({jahr})")

print("\n=== SORTIERT NACH BEWERTUNG ===")
cursor.execute('SELECT titel, bewertung FROM buecher ORDER BY bewertung DESC')
for titel, bewertung in cursor.fetchall():
    print(f"{titel}: {bewertung} ⭐")

print("\n=== STATISTIKEN ===")
cursor.execute('SELECT COUNT(*) FROM buecher')
anzahl = cursor.fetchone()[0]
print(f"Anzahl Bücher: {anzahl}")

cursor.execute('SELECT AVG(bewertung) FROM buecher')
durchschnitt = cursor.fetchone()[0]
print(f"Durchschnittliche Bewertung: {durchschnitt:.2f} ⭐")

# "Die Welle" Bewertung auf 4.5 aktualisieren
print("\n=== UPDATE ===")
cursor.execute('UPDATE buecher SET bewertung = 4.5 WHERE titel = ?', ('Die Welle',))
conn.commit()
print("'Die Welle' wurde auf 4.5 Sterne aktualisiert.")

# "1984" löschen
print("\n=== DELETE ===")
cursor.execute('DELETE FROM buecher WHERE titel = ?', ('1984',))
conn.commit()
print("'1984' wurde aus der Datenbank gelöscht.")

# BONUS: Bücher mit Titel, der mit "D" anfängt
print("\n=== BONUS: LIKE ABFRAGE ===")
cursor.execute('SELECT titel FROM buecher WHERE titel LIKE "D%"')
for titel, in cursor.fetchall():
    print(f"- {titel}")

# BONUS: Spalte "gelesen" hinzufügen
print("\n=== BONUS: ALTER TABLE ===")
try:
    cursor.execute('ALTER TABLE buecher ADD COLUMN gelesen INTEGER DEFAULT 0')
    conn.commit()
    print("Spalte 'gelesen' hinzugefügt!")
except sqlite3.OperationalError:
    print("Spalte 'gelesen' existiert bereits.")

cursor.execute('SELECT * FROM buecher LIMIT 1')
print(f"Neue Spalte: {[description[0] for description in cursor.description]}")

conn.close()

Erklärung: - CREATE TABLE IF NOT EXISTS verhindert Fehler beim mehrfachen Ausführen - executemany() fügt mehrere Datensätze auf einmal ein - Platzhalter ? schützen vor SQL-Injection - ORDER BY ... DESC sortiert absteigend - WHERE jahr > 2000 filtert die Ergebnisse - UPDATE und DELETE verändern bestehende Daten - LIKE "D%" findet alle Titel, die mit “D” anfangen

Aufgabe: init_db.py erweitern

import sqlite3
from datetime import datetime

def init_db(db_name='mydatabase.db'):
    conn = sqlite3.connect(db_name)
    cursor = conn.cursor()

    # schema.sql lesen und ausführen
    with open('schema.sql', 'r') as f:
        cursor.executescript(f.read())

    # Testdaten für buecher
    cursor.execute('''
        INSERT INTO buecher (titel, autor, jahr, bewertung)
        VALUES (?, ?, ?, ?)
    ''', ('Der Hobbit', 'J.R.R. Tolkien', 1937, 4.8))

    cursor.execute('''
        INSERT INTO buecher (titel, autor, jahr, bewertung)
        VALUES (?, ?, ?, ?)
    ''', ('Tschick', 'Wolfgang Herrndorf', 2010, 4.7))

    # Testdaten für notizen
    cursor.execute('''
        INSERT INTO notizen (titel, inhalt, wichtig)
        VALUES (?, ?, ?)
    ''', ('Python lernen', 'SQL-Abfragen üben', 1))

    cursor.execute('''
        INSERT INTO notizen (titel, inhalt, wichtig)
        VALUES (?, ?, ?)
    ''', ('Einkaufen', 'Milch, Brot, Käse', 0))

    conn.commit()
    conn.close()
    print(f"✓ Datenbank {db_name} wurde initialisiert!")

if __name__ == '__main__':
    init_db()

Tipp: Wenn du init_db.py mehrfach ausführst, werden doppelte Daten eingefügt. Du kannst die alte Datenbank mit rm mydatabase.db löschen.

Erweiterte Bücherverwaltung (Flask)

Aufgabe: Suchfunktion

app.py - Suchfunktion

from flask import Flask, render_template, request
import sqlite3

app = Flask(__name__)

def get_db_connection():
    conn = sqlite3.connect('buecher.db')
    conn.row_factory = sqlite3.Row
    return conn

@app.route('/')
def buecher():
    conn = get_db_connection()
    cursor = conn.cursor()

    # Suchterm aus der Query abholen
    search_term = request.args.get('q', '').strip()

    if search_term:
        # Mit LIKE nach Titeln und Autoren suchen
        query = 'SELECT * FROM buecher WHERE titel LIKE ? OR autor LIKE ?'
        cursor.execute(query, (f'%{search_term}%', f'%{search_term}%'))
    else:
        # Alle Bücher wenn nichts gesucht wird
        cursor.execute('SELECT * FROM buecher')

    buecher_liste = cursor.fetchall()

    # Statistiken berechnen
    cursor.execute('SELECT COUNT(*) FROM buecher')
    gesamt = cursor.fetchone()[0]

    cursor.execute('SELECT AVG(bewertung) FROM buecher')
    durchschnitt = cursor.fetchone()[0] or 0

    conn.close()

    return render_template(
        'buecher.html',
        buecher=buecher_liste,
        search_term=search_term,
        gesamt=gesamt,
        durchschnitt=round(durchschnitt, 2)
    )

if __name__ == '__main__':
    app.run(debug=True)

app.py

from flask import Flask, render_template, request, redirect, flash
import sqlite3
from datetime import datetime

app = Flask(__name__)
app.secret_key = 'dein-geheimnis-hier'

def get_db_connection():
    conn = sqlite3.connect('gaestebuch.db')
    conn.row_factory = sqlite3.Row
    return conn

@app.route('/')
def gaestebuch():
    """Zeige alle Gästebucheinträge"""
    conn = get_db_connection()
    cursor = conn.cursor()
    cursor.execute('SELECT * FROM gaestebuch ORDER BY erstellt DESC')
    eintraege = cursor.fetchall()
    conn.close()

    return render_template('gaestebuch.html', eintraege=eintraege)

@app.route('/eintragen', methods=['GET', 'POST'])
def gaestebuch_eintragen():
    """Neuen Eintrag hinzufügen"""
    if request.method == 'POST':
        name = request.form.get('name', '').strip()
        nachricht = request.form.get('nachricht', '').strip()

        # Validierung
        if not name:
            flash('⚠️ Bitte gib deinen Namen ein!', 'warning')
            return redirect('/eintragen')

        if not nachricht:
            flash('⚠️ Bitte schreib eine Nachricht!', 'warning')
            return redirect('/eintragen')

        if len(nachricht) < 3:
            flash('⚠️ Die Nachricht muss mindestens 3 Zeichen lang sein!', 'warning')
            return redirect('/eintragen')

        # In Datenbank speichern
        conn = get_db_connection()
        cursor = conn.cursor()
        cursor.execute(
            'INSERT INTO gaestebuch (name, nachricht) VALUES (?, ?)',
            (name, nachricht)
        )
        conn.commit()
        conn.close()

        flash('✅ Dein Eintrag wurde gespeichert!', 'success')
        return redirect('/')  # Post-Redirect-Get Pattern

    return render_template('gaestebuch_form.html')

@app.route('/loeschen/<int:id>', methods=['POST'])
def gaestebuch_loeschen(id):
    """Eintrag löschen"""
    conn = get_db_connection()
    cursor = conn.cursor()

    # Sicherstellen, dass der Eintrag existiert
    cursor.execute('SELECT * FROM gaestebuch WHERE id = ?', (id,))
    eintrag = cursor.fetchone()

    if eintrag:
        cursor.execute('DELETE FROM gaestebuch WHERE id = ?', (id,))
        conn.commit()
        flash('🗑️ Eintrag gelöscht!', 'info')

    conn.close()
    return redirect('/')

if __name__ == '__main__':
    app.run(debug=True)
