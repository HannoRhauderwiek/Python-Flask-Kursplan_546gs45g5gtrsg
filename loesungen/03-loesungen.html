Übung 1: Taschenrechner

Ein einfacher Taschenrechner mit HTML-Form und Flask-Verarbeitung!

Python-Code (app.py):

from flask import Flask, render_template, request

app = Flask(__name__)

@app.route('/')
def taschenrechner():
    return render_template('taschenrechner.html')

@app.route('/berechnen', methods=['POST'])
def berechnen():
    try:
        zahl1 = float(request.form.get('zahl1', 0))
        zahl2 = float(request.form.get('zahl2', 0))
        operator = request.form.get('operator', '+')

        if operator == '+':
            result = zahl1 + zahl2
        elif operator == '-':
            result = zahl1 - zahl2
        elif operator == '*':
            result = zahl1 * zahl2
        elif operator == '/':
            if zahl2 == 0:
                return render_template('taschenrechner.html',
                                     error='Division durch 0 ist nicht möglich! 😱')
            result = zahl1 / zahl2
        else:
            result = 0

        return render_template('taschenrechner.html',
                             zahl1=zahl1,
                             zahl2=zahl2,
                             operator=operator,
                             result=result)

    except ValueError:
        return render_template('taschenrechner.html',
                             error='Bitte gib gültige Zahlen ein!')

if __name__ == '__main__':
    app.run(debug=True)

Wichtige Punkte: - method='POST' ist wichtig für Formulare (nicht GET!) - request.form.get() holt die Daten aus dem Formular - Mit try...except fangen wir Fehler ab (z.B. wenn der Nutzer Buchstaben eingibt) - Division durch 0 wird speziell behandelt - Die Ergebnisse werden zurück in die Template mit render_template(..., result=result) übergeben - if result is not none zeigt das Ergebnis nur, wenn es eins gibt

Übung 2: Login-Seite

Eine einfache Login-Seite mit Passwort-Check und Nachricht!

Python-Code (app.py):

from flask import Flask, render_template, request

app = Flask(__name__)

GEHEIMES_PASSWORT = 'geheim123'

@app.route('/')
def login():
    return render_template('login.html')

@app.route('/check-login', methods=['POST'])
def check_login():
    benutzername = request.form.get('benutzername', '').strip()
    passwort = request.form.get('passwort', '')

    if not benutzername:
        return render_template('login.html', error='Benutzername erforderlich!')

    if passwort == GEHEIMES_PASSWORT:
        return render_template('erfolg.html', benutzername=benutzername)
    else:
        return render_template('login.html',
                             benutzername=benutzername,
                             error='Falsches Passwort! Versuch es erneut.')

if __name__ == '__main__':
    app.run(debug=True)

Übung 3: Feedback-System (Challenge)

Das coolste Projekt! Ein komplettes Feedback-System mit Validierung!

Python-Code (app.py):

from flask import Flask, render_template, request, redirect, url_for, flash

app = Flask(__name__)
app.secret_key = 'dein-geheimnis-hier'

# Speicher für Feedback (In der Praxis würde man eine Datenbank nutzen!)
feedbacks = []

@app.route('/')
def index():
    return render_template('feedback.html', feedbacks=feedbacks)

@app.route('/feedback-senden', methods=['POST'])
def feedback_senden():
    # Daten aus dem Formular holen
    name = request.form.get('name', '').strip()
    email = request.form.get('email', '').strip()
    kategorie = request.form.get('kategorie', '')
    rating = request.form.get('rating', '5')
    nachricht = request.form.get('nachricht', '').strip()

    # Validierung
    fehler = []

    if not name:
        fehler.append('Bitte gib deinen Namen ein.')
    if not email:
        fehler.append('Bitte gib deine E-Mail ein.')
    elif '@' not in email:
        fehler.append('Bitte gib eine gültige E-Mail ein.')
    if not kategorie:
        fehler.append('Bitte wähle eine Kategorie.')
    if len(nachricht) < 10:
        fehler.append('Deine Nachricht muss mindestens 10 Zeichen lang sein.')

    # Wenn Fehler, zurück zum Formular
    if fehler:
        return render_template('feedback.html',
                             error=fehler,
                             feedbacks=feedbacks,
                             name=name,
                             email=email,
                             kategorie=kategorie,
                             rating=rating,
                             nachricht=nachricht)

    # Feedback speichern
    neues_feedback = {
        'name': name,
        'email': email,
        'kategorie': kategorie,
        'rating': int(rating),
        'nachricht': nachricht
    }
    feedbacks.insert(0, neues_feedback)  # Am Anfang einfügen (neuste zuerst)

    # PRG-Pattern: Redirect-Render-Get
    return redirect(url_for('danke'))

@app.route('/danke')
def danke():
    return render_template('danke.html', anzahl=len(feedbacks))

if __name__ == '__main__':
    app.run(debug=True)

Wichtige Konzepte in dieser Lösung:

  • Validierung: Alle Felder werden auf Seite des Servers geprüft
  • PRG-Pattern (Post-Redirect-Get): Nach dem Senden wird zu einer neuen Seite weitergeleitet
  • Fehlerbehandlung: Fehler werden als Liste angezeigt
  • Persistenz: Feedback wird in einer Python-Liste gespeichert (reicht für Demo!)
  • Jinja2-Filter: |length zählt die Feedbacks
  • Loops: Mit range() zeigen wir Sterne an
  • JavaScript: Einfaches Zeichen-Zählen für die Nachricht
